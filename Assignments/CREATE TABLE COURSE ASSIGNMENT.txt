CREATE TABLE Customer(
CustNo VARchar(11) not null,
custname varchar(30) not null,

Internal char(2) not null,
contact  varchar(50) not null,

Zip char(10) not null,
CONSTRAINT CustPk PRIMARY KEY (custNo) );

6. 
CREATE TABLE EVENTREQUEST(
EVENTNO VARCHAR(8) NOT NULL,
DATEHELD VARCHAR(10) NOT NULL,
DATEREQ VARCHAR(10) NOT NULL,
CUSTNO VARCHAR(20) NOT NULL,
FACNO VARCHAR(20) NOT NULL,
DATEAUTH VARCHAR(10),
STATUS varchar(20) NOT NULL,
ESTCOST VARCHAR(10) NOT NULL,
ESTAUDIENCE VARHCAR(20) NOT NULL ,
BUDNO INTEGER,
CONSTRAINT EVENTNO_PK PRIMARY KEY (EVENTNO),
CONSTRAINT CHK_STATUS CHECK(STATUS = 'Pending' or STATUS  = 'Denied' or STATUS = 'Approved')
CONSTRAINT CHK_ESTAUDIANCE CHECK(ESTAUDIENCE > 0)
);

7.
CREATE TABLE EVENTPLAN(
PLANNO VARCHAR(8) NOT NULL,
EVENTNO VARCHAR(8) NOT NULL,
WORKDATE VARCHAR(10) NOT NULL,
NOTES VARCHAR(20),
ACTIVITY VARCHAR(20) NOT NULL,
EMPNO VARCHAR(8) ,
CONSTRAINT PLANNO_PK PRIMARY KEY (PLANNO)
CONSTRAINT EVENTNO_FK FOREIGN KEY (EVENTNO)
 REFERENCES EVENTREQUEST(EVENTNO) );

8.
CREATE TABLE EVENTPLANLINE(
PLANNO VARCHAR(8) NOT NULL,
LINENO VARCHAR(8) NOT NULL,
TIMESTART VARCHAR(8),
TIMEEND VARCHAR(8)
NUMBERFLD VARCHAR(20),
LOCNO VARCHAR(8) NOT NULL,
RESNO VARCHAR(8) NOT NULL,
CONSTRAINT LINENO_PK PRIMARY KEY (LINENO),
CONSTRAINT PLANNO_FK FOREIGN KEY (PLANNO)
     REFERENCES EVENTPLAN(PLANNO) 
CONSTRAINT LOCNO_FK FOREIGN KEY (LOCNO)
     REFERENCES LOCATION(LOCNO)
CONSTRAINT RESNO_FK FOREIGN KEY (RESNO)
     REFERENCES RESOURCETBL(RESNO) 
CONSTRAINT CHK_TIMESTRT_AND_TIMEEND CHECK (TIMESTART < TIMEEND) );
